# API GateWay + 缓存服务器

大规模录制

## 方案一  Nginx + MongoDB

**Nginx** 担任反向代理，接收客户端请求。

**Lua 脚本** 在 Nginx 内部做缓存查询和写入逻辑。

**MongoDB** 存储请求-响应对，支持持久化和高并发查询。

###  nginx 配置

```conf
http {
    lua_package_path "/usr/local/nginx/lua/?.lua;;";
    
    server {
        listen 8080;
        
        location / {
            access_by_lua_file /usr/local/nginx/lua/cache_lookup.lua;
            proxy_pass http://real-service:8081;
            body_filter_by_lua_file /usr/local/nginx/lua/cache_write.lua;
        }
    }
}

```

### lua 查缓存【cache_lookup.lua】

```lua
local cjson = require "cjson"
local mongo = require "resty.mongo"  -- 使用 OpenResty MongoDB 库

local client = mongo:new()
client:set_timeout(1000)
client:connect("mongodb://127.0.0.1:27017")

-- 生成请求 hash
local req_str = ngx.var.request_method .. ngx.var.uri .. ngx.var.args
local req_hash = ngx.md5(req_str)

-- 查询 MongoDB
local doc = client:get_by_id("mockdb", "requests", req_hash)
if doc then
    ngx.status = doc.response.status
    for k, v in pairs(doc.response.headers) do
        ngx.header[k] = v
    end
    ngx.say(doc.response.body)
    ngx.exit(ngx.HTTP_OK)
end
-- 未命中，继续 proxy_pass

```

### lua 写入缓存【cache_write.lua】

```lua
local cjson = require "cjson"
local mongo = require "resty.mongo"

local client = mongo:new()
client:set_timeout(1000)
client:connect("mongodb://127.0.0.1:27017")

-- 生成请求 hash
local req_str = ngx.var.request_method .. ngx.var.uri .. ngx.var.args
local req_hash = ngx.md5(req_str)

-- 获取响应 body
local resp_body = ngx.arg[1]

-- 写入 MongoDB
client:insert("mockdb", "requests", {
    _id = req_hash,
    method = ngx.var.request_method,
    url = ngx.var.uri,
    query = ngx.var.args,
    response = {
        status = ngx.status,
        headers = ngx.resp.get_headers(),
        body = resp_body
    },
    createdAt = ngx.time()
})

```
