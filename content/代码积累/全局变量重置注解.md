import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target(ElementType.FIELD)
@Retention(RetentionPolicy.RUNTIME)
public @interface ResetGlobalVariable {
    String initBooleanValue() default "";
}

import org.apache.commons.lang3.StringUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.reflect.Field;
import java.util.Collection;
import java.util.Collections;
import java.util.List;
import java.util.Map;

public class RestGlobalVariableParser {

    private static final Logger log = LoggerFactory.getLogger(RestGlobalVariableParser.class);

    /**
     * 清理并重建所有带有 @ResetGlobalVariable 注解的字段
     *
     * @param context 上下文对象（如 ComputeContext）
     */
    public static void clearFields(Object context) {
        Class<?> clazz = context.getClass();
        Field[] fields = clazz.getDeclaredFields();

        for (Field field : fields) {
            if (!field.isAnnotationPresent(ResetGlobalVariable.class)) {
                continue;
            }

            try {
                field.setAccessible(true);
                ResetGlobalVariable annotation = field.getAnnotation(ResetGlobalVariable.class);
                String booleanStr = annotation.initBooleanValue();
                if (StringUtils.isNotEmpty(booleanStr)) {
                    field.set(context, Boolean.parseBoolean(booleanStr));
                    log.debug("Reset field using initBooleanValue: {}", field.getName());
                } else if (isCollectionOrMap(fieldType)) {
                    Class<?> fieldType = field.getType();
                    Object emptyCollection = createEmptyCollectionOrMap(fieldType);
                    field.set(context, emptyCollection);
                    log.debug("Reset collection/map field: {}", field.getName());
                }
            } catch (Exception e) {
                log.error("Failed to reset field annotated with @ClearGloable", e);
            }
        }
    }

    /**
     * 判断是否为集合或 Map 类型
     */
    private static boolean isCollectionOrMap(Class<?> type) {
        return Collection.class.isAssignableFrom(type) || Map.class.isAssignableFrom(type);
    }

    /**
     * 创建空集合或 Map 实例
     */
    private static Object createEmptyCollectionOrMap(Class<?> type) {
        if (type.isAssignableFrom(List.class)) {
            return Collections.emptyList();
        } else if (type.isAssignableFrom(Map.class)) {
            return Collections.emptyMap();
        } else {
            return Collections.emptySet();
        }
    }
}

    @ResetGlobalVariable(initBooleanValue = "false")

        ComputeContext computeContext = new ComputeContext();
        computeContext.getDataList().add("1");
        computeContext.getResult1().add("1");
