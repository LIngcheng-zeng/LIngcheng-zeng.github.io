    function parseCSVLine(text) {
        if (!text) return [];
        const res = []; let start = 0, inQ = false;
        for (let i = 0; i < text.length; i++) {
            if (text[i] === '"') inQ = !inQ;
            else if (text[i] === ',' && !inQ) {
                let f = text.substring(start, i).trim();
                if (f.startsWith('"') && f.endsWith('"')) f = f.slice(1, -1);
                res.push(f); start = i + 1;
            }
        }
        let last = text.substring(start).trim();
        if (last.startsWith('"') && last.endsWith('"')) last = last.slice(1, -1);
        res.push(last);
        return res;
    }

    function parseCSV(content, mapping) {
        const lines = content.trim().split(/\r?\n/);
        if (lines.length < 2) throw new Error("文件无效");
        const headers = parseCSVLine(lines[0].replace(/^\uFEFF/, '')).map(h => h.trim());
        const norm = {
            cur: mapping.currentId.toLowerCase().trim(),
            par: mapping.parentId.toLowerCase().trim(),
            aPre: mapping.aInfo.toLowerCase().trim(),
            zPre: mapping.zInfo.toLowerCase().trim()
        };

        const iCur = headers.findIndex(h => h.toLowerCase() === norm.cur);
        const iPar = headers.findIndex(h => h.toLowerCase() === norm.par);

        if (iCur === -1 || iPar === -1) {
            throw new Error(`缺少必要列。查找："${mapping.currentId}" (索引:${iCur}), "${mapping.parentId}" (索引:${iPar})`);
        }

        const data = [];
        for (let i = 1; i < lines.length; i++) {
            if (!lines[i].trim()) continue;
            const cols = parseCSVLine(lines[i]);
            if (cols.length <= Math.max(iCur, iPar)) continue;

            const getInfos = (pre) => {
                const obj = {};
                headers.forEach((h, idx) => {
                    if (h.toLowerCase().startsWith(pre)) {
                        let k = h.substring(pre.length).trim().replace(/^[-_:]\s*/, '');
                        obj[k || 'Info'] = cols[idx] || '未知';
                    }
                });
                return Object.keys(obj).length ? obj : { 'Info': '无详细信息' };
            };

            // 强制字符串化并去除空格
            const rawId = String(cols[iCur]).trim();
            const rawParentId = String(cols[iPar]).trim();

            data.push({
                id: rawId,
                parentId: rawParentId,
                aInfo: getInfos(norm.aPre),
                zInfo: getInfos(norm.zPre)
            });
        }
        return data;
    }

    function getSignature(infoObj) {
        if (!infoObj) return '';
        const port = infoObj['端口'] || infoObj['Port'] || '';
        const ip = infoObj['IP 地址'] || infoObj['IP'] || '';
        const device = infoObj['设备类型'] || infoObj['Device'] || '';
        if (port || ip || device) return `${device}|${port}|${ip}`.toLowerCase().trim();
        const genericInfo = infoObj['信息'] || infoObj['Info'] || infoObj['Content'] || '';
        if (genericInfo) return String(genericInfo).toLowerCase().trim();
        return Object.values(infoObj).join('|').toLowerCase().trim();
    }

    function transformToG6Data(rawData) {
        const nodeMap = new Map();
        const nodes = [];
        const edges = [];
        const depthMap = new Map();

        // 1. 数据清洗与节点注册
        const cleanData = rawData.map(item => ({
            ...item,
            id: String(item.id).trim(),
            parentId: String(item.parentId).trim()
        }));

        // 收集所有出现的 ID
        const allIds = new Set();
        cleanData.forEach(item => {
            allIds.add(item.id);
            if (item.parentId && item.parentId !== '0' && item.parentId !== '') {
                allIds.add(item.parentId);
            }
        });

        // 创建虚拟根节点 '0'
        const hasRootChildren = cleanData.some(item => item.parentId === '0' || item.parentId === '');
        if (hasRootChildren || allIds.has('0')) {
            if (!nodeMap.has('0')) {
                nodeMap.set('0', {
                    id: '0', parentId: '',
                    aInfo: { '设备类型': '核心根节点', '端口': 'ROOT', 'IP 地址': '0.0.0.0' },
                    zInfo: {}, isPlaceholder: false, definedInData: false
                });
            }
        }

        // 注册所有实际数据节点
        cleanData.forEach(item => {
            if (!nodeMap.has(item.id)) {
                nodeMap.set(item.id, { ...item, isPlaceholder: false, definedInData: true });
            }
        });

        // 2. 【关键修复】直接构建父子连线，不依赖 BFS 结果
        // 只要 CSV 里写了 parentId，且该 parentId 对应的节点存在（或为虚拟根），就连线
        cleanData.forEach(item => {
            const target = item.id;
            let source = item.parentId;

            // 规范化源节点：如果是 0 或 空，指向虚拟根 '0'
            if (source === '0' || source === '') {
                source = '0';
            }

            // 检查源节点是否存在于 nodeMap 中
            // 如果源节点在 CSV 中作为“当前节点”出现过，它一定在 nodeMap 中
            // 如果源节点只作为“父节点”出现过但没作为“当前节点”出现，我们需要创建一个占位符或者确保它被注册了
            if (!nodeMap.has(source)) {
                // 这种情况理论上不应发生，因为上面 allIds 收集了所有 parentId
                // 但为了保险，创建一个隐式节点
                nodeMap.set(source, {
                    id: source, parentId: '', aInfo: {'Info': '隐式节点'}, zInfo: {},
                    isPlaceholder: true, definedInData: false
                });
            }

            if (source !== target) { // 避免自环
                const sNode = nodeMap.get(source);
                const tNode = nodeMap.get(target);

                const getLabel = (n) => {
                    if(!n) return '';
                    if(n.aInfo && n.aInfo['端口']) return n.aInfo['端口'];
                    const k = Object.keys(n.aInfo||{}).find(key => key.includes('信息')||key.includes('Info'));
                    return k ? n.aInfo[k] : n.id;
                };

                const label = source === '0' ? `ROOT ➜ ${getLabel(tNode)}` : `${getLabel(sNode)} ➜ ${getLabel(tNode)}`;

                edges.push({
                    source, target, type: 'quadratic', label,
                    model: { sourceNode: sNode, targetNode: tNode, relationType: source==='0'?'根连接':'父子连接' },
                    style: {
                        stroke: source === '0' ? '#722ed1' : '#595959',
                        lineWidth: source === '0' ? 3 : 2,
                        endArrow: { path: 'M 0,0 L 8,-4 L 8,4 Z', fill: source === '0' ? '#722ed1' : '#595959' },
                        labelCfg: { autoRotate: true, style: { fontSize: 10, fill: '#666', background: { fill: '#fff', padding: [2, 4], radius: 2 } } }
                    }
                });
            }
        });

        // 3. 计算深度 (仅用于布局，不影响连线)
        // 初始化根节点深度
        nodeMap.forEach((node, id) => {
            if (node.parentId === '0' || node.parentId === '' || !nodeMap.has(node.parentId)) {
                depthMap.set(id, 0);
            }
        });
        if(nodeMap.has('0')) depthMap.set('0', 0);

        let changed = true;
        while(changed) {
            changed = false;
            cleanData.forEach(item => {
                const pid = item.parentId === '0' || item.parentId === '' ? '0' : item.parentId;
                if (depthMap.has(pid) && !depthMap.has(item.id)) {
                    depthMap.set(item.id, depthMap.get(pid) + 1);
                    changed = true;
                }
            });
        }

        // 4. 构建 Nodes 数组
        nodeMap.forEach((item, id) => {
            let label = id;
            if (!item.isPlaceholder && item.aInfo) {
                const k = Object.keys(item.aInfo).find(key => key.includes('端口')||key.includes('Info')||key.includes('信息'));
                if(k && item.aInfo[k] !== id) label = `${id}\n${item.aInfo[k]}`;
            }
            if(id === '0') label = '核心根节点\n(ROOT)';

            nodes.push({
                id, label, type: 'circle',
                size: id === '0' ? 60 : 50,
                style: {
                    fill: id === '0' ? '#722ed1' : '#5470c6',
                    stroke: '#fff', lineWidth: 2, shadowColor: 'rgba(0,0,0,0.2)', shadowBlur: 10
                },
                labelCfg: { style: { fontSize: id==='0'?14:12, fill: '#333', fontWeight: 'bold' }, position: 'bottom', offset: 5 },
                aInfo: item.aInfo, zInfo: item.zInfo, depth: depthMap.get(id) || 0
            });
        });

        // 5. 构建业务连线 (同层匹配)
        const levels = {};
        nodes.forEach(n => {
            const d = n.depth || 0;
            if(!levels[d]) levels[d] = [];
            levels[d].push(n);
        });

        Object.values(levels).forEach(levelNodes => {
            for(let i=0; i<levelNodes.length; i++) {
                for(let j=i+1; j<levelNodes.length; j++) {
                    const n1 = levelNodes[i];
                    const n2 = levelNodes[j];
                    if(n1.id==='0' || n2.id==='0') continue;

                    const s1A = getSignature(n1.aInfo), s1Z = getSignature(n1.zInfo);
                    const s2A = getSignature(n2.aInfo), s2Z = getSignature(n2.zInfo);

                    let match = false, type = '', desc = '';
                    if(s1A && s2A && s1A===s2A && s1Z && s2Z && s1Z===s2Z) { match=true; type='complete'; desc='完全匹配'; }
                    else if(s1A && s2Z && s1A===s2Z) { match=true; type='cross'; desc='A-Z 交叉'; }
                    else if(s1Z && s2A && s1Z===s2A) { match=true; type='cross'; desc='Z-A 交叉'; }

                    if(match) {
                        edges.push({
                            source: n1.id, target: n2.id, type: 'line', label: `${n1.id} ⇋ ${n2.id}`,
                            model: { relationType: '业务关联', matchReason: desc },
                            style: {
                                stroke: type==='cross'?'#13c2c2':'#fa8c16', lineWidth: 1.5,
                                lineDash: type==='cross'?[2,2]:[5,5], opacity: 0.8,
                                labelCfg: { style: { fontSize: 9, fill: type==='cross'?'#08979c':'#d46b08', background: {fill:'rgba(255,255,255,0.9)', padding:[1,3], radius:2} } }
                            }
                        });
                    }
                }
            }
        });

        console.log(`构建完成：${nodes.length} 节点，${edges.length} 边`);
        console.log('深度图:', Object.fromEntries(depthMap));
        return { nodes, edges };
    }

    function initGraph() {
        const container = document.getElementById('container');
        if (!container) return;
        graph = new G6.Graph({
            container: 'container', width: container.scrollWidth, height: container.scrollHeight || 600,
            fitView: true, fitViewPadding: 50,
            modes: { default: ['drag-canvas', 'zoom-canvas', 'drag-node', 'activate-relations'] },
            layout: { type: 'dagre', rankdir: 'TB', nodesep: 50, ranksep: 100, controlPoints: true },
            plugins: [{
                type: 'tooltip', offsetX: 10, offsetY: 10, itemTypes: ['node', 'edge'], trigger: 'mouseenter',
                getContent: (e) => {
                    const m = e.item.getModel();
                    if(e.item.getType()==='edge') return `<div class="g6-tooltip"><strong>${m.model?.relationType}</strong><br/>${m.label}</div>`;
                    const renderTable = (info, title, cls) => {
                        if(!info) return '';
                        let h = `<span class="tooltip-title ${cls}">${title}</span><table class="tooltip-table">`;
                        for(const [k,v] of Object.entries(info)) h+=`<tr><td class="tooltip-key">${k}</td><td class="tooltip-val">${v}</td></tr>`;
                        return h+'</table>';
                    };
                    return `<div class="g6-tooltip" style="width:250px"><strong>${m.label.split('\n')[0]}</strong><br/>ID: ${m.id}<br/>${renderTable(m.aInfo,'A 端','title-a')}${renderTable(m.zInfo,'Z 端','title-z')}</div>`;
                }
            }]
        });

        graph.on('edge:click', (evt) => {
            const edge = evt.item; if(!edge) return;
            const m = edge.getModel();
            const popover = document.getElementById('edgeDetailPopover');
            document.getElementById('edgeTitle').innerText = `${m.model?.relationType}: ${m.source} ⇋ ${m.target}`;
            document.getElementById('edgeContent').innerHTML = `<div style="padding:5px 0">${m.model?.matchReason || ''}</div>`;
            popover.style.left = `${evt.clientX+20}px`; popover.style.top = `${evt.clientY+20}px`; popover.style.display = 'block';
        });
        graph.on('canvas:click', () => document.getElementById('edgeDetailPopover').style.display='none');
        window.onresize = () => { if(graph) graph.changeSize(container.scrollWidth, container.scrollHeight); };
    }

    function renderData(data) {
        if(!graph) initGraph();
        const g6Data = transformToG6Data(data);
        graph.data(g6Data); graph.render(); graph.fitView(50);
        flatData = data;
        setStatus(`成功：${data.length} 记录`, 'success');
    }

    function processFile(content) {
        if(!content) return;
        const mapping = {
            currentId: dom.configs.currentId.value.trim(),
            parentId: dom.configs.parentId.value.trim(),
            aInfo: dom.configs.aInfo.value.trim(),
            zInfo: dom.configs.zInfo.value.trim()
        };
        setLoading(true);
        setTimeout(() => {
            try {
                const data = parseCSV(content, mapping);
                renderData(data); setLoading(false); lastContent = content;
            } catch(e) {
                setStatus(`错误：${e.message}`, 'error'); setLoading(false);
            }
        }, 50);
    }

    dom.fileInput.addEventListener('change', e => {
        const file = e.target.files[0];
        if(!file || !file.name.endsWith('.csv')) return setStatus('仅支持 CSV', 'error');
        const reader = new FileReader();
        reader.onload = evt => processFile(evt.target.result);
        reader.readAsText(file);
    });

    Object.values(dom.configs).forEach(input => {
        input.addEventListener('input', () => { if(lastContent) processFile(lastContent); });
    });

    initGraph();
    setStatus('就绪', 'normal');
</script>
