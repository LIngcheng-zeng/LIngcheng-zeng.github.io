    function doSearch() {
        const searchText = dom.searchInput.value.trim();
        if (!searchText || !graph) { setStatus('请输入搜索内容', 'normal'); return; }
        
        const nodes = graph.getNodes();
        let found = null;
        for (let i = 0; i < nodes.length; i++) {
            const m = nodes[i].getModel();
            if (m.id.includes(searchText) || (m.label && m.label.includes(searchText))) {
                found = m.id;
                break;
            }
        }
        if (found) {
            graph.focusItem(found, true, { easing: 'easeCubic', duration: 500 });
            setStatus(`找到: ${found}`, 'success');
        } else {
            setStatus(`未找到: "${searchText}"`, 'error');
        }
    }

    function resetConfig() {
        dom.configs.currentId.value = '当前节点';
        dom.configs.parentId.value = '父节点';
        dom.configs.aInfo.value = 'A 端';
        dom.configs.zInfo.value = 'Z 端';

    function setupEventListeners() {
        // File input

                reader.onload = evt => processFile(evt.target.result);

            });

        // Config inputs
        Object.values(dom.configs).forEach(input => {
            if (input) {
                input.addEventListener('input', () => { if (lastContent) processFile(lastContent); });
            }
        });
    }

    function expandAll() {
        if (!lazyLoadMode || !graph) return;
        
        const allNodes = [], allEdges = [];
        nodeInfoMap.forEach((_, id) => {
            const model = createNodeModel(id);
            if (model) allNodes.push(model);
        });
        
        flatData.forEach(item => {
            if (item.parentId && item.parentId !== '0') {
                allEdges.push({

                    target: item.id,
                    type: 'quadratic',
                    style: { stroke: '#595959', lineWidth: 1.5, endArrow: { path: 'M 0,0 L 6,-3 L 6,3 Z', fill: '#595959' } }

        loadedNodes.clear();
        nodeInfoMap.forEach((_, id) => loadedNodes.add(id));
        
        graph.read({ nodes: allNodes, edges: allEdges });
        graph.fitView(60);
        updateStats();
        setStatus(`已展开全部 ${loadedNodes.size} 个节点`, 'success');
    }

    function collapseToRoot() {
        if (!lazyLoadMode || !graph) return;
        
        const rootChildren = parentChildMap.get('0') || [];
        const nodes = [], edges = [];
        
        loadedNodes.clear();
        loadedNodes.add('0');
        
        const rootModel = createNodeModel('0');
        if (rootModel) nodes.push(rootModel);
        
        rootChildren.forEach(child => {
            loadedNodes.add(child.id);
            const m = createNodeModel(child.id);
            if (m) {
                nodes.push(m);
                edges.push({
                    source: '0', target: child.id, type: 'quadratic',
                    style: { stroke: '#722ed1', lineWidth: 2, endArrow: { path: 'M 0,0 L 6,-3 L 6,3 Z', fill: '#722ed1' } }
                });
            }
        });
        
        graph.read({ nodes, edges });
        graph.fitView(60);
        updateStats();
        setStatus(`已收起，显示 ${loadedNodes.size} 个节点`, 'success');
    }

        else if (type === 'success') dom.status.innerHTML = `<span class="msg-success">✅ ${msg}</span>`;
        else dom.status.innerText = msg;
    }

    function updateStats() {
        if (dom.statsPanel) {
            dom.statsPanel.style.display = 'block';

            document.getElementById('statTotal').textContent = nodeInfoMap.size;

    // ========== CSV Parsing ==========
    function parseCSVLine(text) {
        if (!text) return [];
        const res = [], parts = [];
        let start = 0, inQ = false;
        for (let i = 0; i < text.length; i++) {
            if (text[i] === '"') inQ = !inQ;
            else if (text[i] === ',' && !inQ) {
                let f = text.substring(start, i).trim();
                if (f.startsWith('"') && f.endsWith('"')) f = f.slice(1, -1);
                parts.push(f);
                start = i + 1;
            }
        }
        let last = text.substring(start).trim();
        if (last.startsWith('"') && last.endsWith('"')) last = last.slice(1, -1);
        parts.push(last);
        return parts;
    }

    function parseCSV(content, mapping) {
        const lines = content.trim().split(/\r?\n/);
        if (lines.length < 2) throw new Error("文件无效");

        const headers = parseCSVLine(lines[0].replace(/^\uFEFF/, '')).map(h => h.trim());
        
        const findCol = (target) => {
            let idx = headers.findIndex(h => h.toLowerCase().trim() === target.toLowerCase().trim());
            if (idx !== -1) return idx;
            idx = headers.findIndex(h => h.toLowerCase().trim().includes(target.toLowerCase().trim()));
            if (idx !== -1) return idx;
            if (!target.includes(' ')) {
                idx = headers.findIndex(h => h.toLowerCase().trim().startsWith(target.toLowerCase().trim()));
            }
            return idx;
        };

        const iCur = findCol(mapping.currentId);
        const iPar = findCol(mapping.parentId);

        if (iCur === -1 || iPar === -1) {
            throw new Error(`缺少列: 当前节点="${mapping.currentId}", 父节点="${mapping.parentId}"`);
        }

        const data = [];
        for (let i = 1; i < lines.length; i++) {
            if (!lines[i].trim()) continue;
            const cols = parseCSVLine(lines[i]);
            if (cols.length <= Math.max(iCur, iPar)) continue;

            const getInfo = (prefix) => {
                const obj = {};
                headers.forEach((h, idx) => {
                    const hl = h.toLowerCase().trim();
                    if (hl.startsWith(prefix.toLowerCase().trim()) || hl.includes(prefix.toLowerCase().trim())) {
                        let k = h.substring(prefix.length).trim().replace(/^[-_:]\s*/, '');
                        if (!k) k = '信息';
                        obj[k.trim()] = cols[idx] || '-';
                    }
                });
                return Object.keys(obj).length ? obj : null;
            };

            const rawId = String(cols[iCur] || '').trim();
            const rawParentId = String(cols[iPar] || '').trim();
            if (!rawId) continue;

            data.push({
                id: rawId,
                parentId: rawParentId || '0',
                aInfo: getInfo(mapping.aInfo),
                zInfo: getInfo(mapping.zInfo)
            });
        }
        return data;
    }

    // ========== Data Prep ==========
    function prepareData(rawData) {
        nodeInfoMap = new Map();
        parentChildMap.clear();
        
        isLargeDataset = rawData.length > LARGE_DATASET_THRESHOLD;
        totalRecords = rawData.length;

        const cleanData = rawData.map(item => ({
            id: String(item.id).trim(),
            parentId: String(item.parentId).trim()
        }));

        // Collect all IDs
        const allIds = new Set();
        cleanData.forEach(item => {
            allIds.add(item.id);
            const pid = item.parentId || '0';
            if (pid !== '0') allIds.add(pid);
        });
        if (cleanData.some(d => !d.parentId || d.parentId === '0')) allIds.add('0');

        // Build maps
        allIds.forEach(id => {
            const item = cleanData.find(d => d.id === id);
            const isRoot = id === '0';
            nodeInfoMap.set(id, {
                id,
                parentId: item ? item.parentId : (isRoot ? '' : '0'),
                aInfo: item ? item.aInfo : (isRoot ? { '类型': '根节点' } : { '类型': '隐式节点' }),
                zInfo: item ? item.zInfo : {}
            });
            parentChildMap.set(id, []);
        });

        cleanData.forEach(item => {
            const pid = item.parentId || '0';
            if (parentChildMap.has(pid)) {
                parentChildMap.get(pid).push(item);
            }
        });

        console.log(`准备完成: ${allIds.size} 节点`);
    }

    function createNodeModel(id) {
        const node = nodeInfoMap.get(id);
        if (!node) return null;

        let label = id;
        if (node.aInfo) {
            const key = Object.keys(node.aInfo).find(k => k.includes('信息') || k.includes('端口'));
            if (key && node.aInfo[key] !== id && node.aInfo[key] !== '-') {
                label = `${id}\n${node.aInfo[key]}`;
            }
        }
        if (id === '0') label = '根节点\n(ROOT)';

        const size = id === '0' ? 50 : 40;
        const fSize = id === '0' ? 12 : 10;
        const hasChildren = (parentChildMap.get(id) || []).length > 0;

        return {
            id,
            label,
            type: 'circle',
            size,
            style: {
                fill: id === '0' ? '#722ed1' : '#5470c6',
                stroke: '#fff',
                lineWidth: 2,
                cursor: hasChildren ? 'pointer' : 'default'
            },
            labelCfg: {
                style: { fontSize: fSize, fill: '#333', fontWeight: 'bold' },
                position: 'bottom',
                offset: 4
            },
            aInfo: node.aInfo,
            zInfo: node.zInfo,
            hasChildren
        };
    }

    // ========== G6 Graph ==========
    function initGraph() {
        const container = document.getElementById('container');
        if (!container) return;

        graph = new G6.Graph({
            container: 'container',
            width: container.scrollWidth || 800,
            height: container.scrollHeight || 600,
            fitView: true,
            fitViewPadding: 60,
            animate: true,
            modes: { default: ['drag-canvas', 'zoom-canvas', 'drag-node'] },
            layout: { type: 'dagre', rankdir: 'TB', nodesep: 40, ranksep: 70 },
            plugins: [new G6.Tooltip({
                offsetX: 10,
                offsetY: 10,
                itemTypes: ['node'],
                getContent: (e) => {
                    const m = e.item.getModel();
                    const renderInfo = (info, title) => {
                        if (!info) return '';
                        let h = `<span class="tooltip-title">${title}</span><table class="tooltip-table">`;
                        for (const [k, v] of Object.entries(info)) {
                            if (v && v !== '-') h += `<tr><td class="tooltip-key">${k}</td><td class="tooltip-val">${v}</td></tr>`;
                        }
                        return h + '</table>';
                    };
                    return `<div class="g6-tooltip" style="width:200px"><strong>${m.label.split('\n')[0]}</strong>${renderInfo(m.aInfo, 'A端')}${renderInfo(m.zInfo, 'Z端')}</div>`;
                }
            })]
        });

        // Node click - expand/collapse
        graph.on('node:click', (evt) => {
            const node = evt.item;
            if (!node) return;
            
            const nodeId = node.getModel().id;
            const children = parentChildMap.get(nodeId) || [];
            
            if (children.length === 0) {
                setStatus(`节点 "${nodeId}" 没有子节点`, 'normal');
                return;
            }
            
            // Check loaded children
            const unloaded = children.filter(c => !loadedNodes.has(c.id));
            
            if (unloaded.length === 0) {
                // Collapse - remove descendants
                const toRemove = new Set();
                const removeDescendants = (pid) => {
                    const kids = parentChildMap.get(pid) || [];
                    kids.forEach(c => {
                        if (loadedNodes.has(c.id)) {
                            toRemove.add(c.id);
                            removeDescendants(c.id);
                        }
                    });
                };
                removeDescendants(nodeId);
                
                toRemove.forEach(id => {
                    loadedNodes.delete(id);
                    const item = graph.findById(id);
                    if (item) graph.removeItem(item);
                });
                
                graph.layout();
                graph.fitView(60);
                updateStats();
                setStatus(`已收起 "${nodeId}" 的子节点`, 'success');
                return;
            }
            
            // Expand - add children
            const newNodes = [], newEdges = [];
            
            unloaded.forEach(child => {
                loadedNodes.add(child.id);
                const m = createNodeModel(child.id);
                if (m) {
                    newNodes.push(m);
                    newEdges.push({
                        source: nodeId,
                        target: child.id,
                        type: 'quadratic',
                        style: { stroke: '#595959', lineWidth: 1.5, endArrow: { path: 'M 0,0 L 6,-3 L 6,3 Z', fill: '#595959' } }
                    });
                }
            });
            
            newNodes.forEach(n => graph.addItem('node', n));
            newEdges.forEach(e => graph.addItem('edge', e));
            graph.layout();
            graph.fitView(60);
            updateStats();
            
            setStatus(`展开 ${newNodes.length} 个子节点 (${loadedNodes.size}/${nodeInfoMap.size})`, 'success');
        });

        // Edge click
        graph.on('edge:click', (evt) => {
            const edge = evt.item;
            if (!edge) return;
            const m = edge.getModel();
            const pop = document.getElementById('edgePopover');
            document.getElementById('edgeTitle').textContent = `${m.source} → ${m.target}`;
            document.getElementById('edgeContent').innerHTML = `<div class="edge-row"><span class="edge-label">类型:</span><span class="edge-val">父子</span></div>`;
            pop.style.left = (evt.clientX + 15) + 'px';
            pop.style.top = (evt.clientY + 15) + 'px';
            pop.style.display = 'block';
        });

        graph.on('canvas:click', closeEdgePopover);
        window.onresize = () => { if (graph) graph.changeSize(container.scrollWidth, container.scrollHeight); };
    }

    // ========== Render ==========
    function renderData(data) {
        lazyLoadMode = dom.lazyLoadCheckbox ? dom.lazyLoadCheckbox.checked : false;
        prepareData(data);
        
        if (!graph) initGraph();

        if (lazyLoadMode) {
            // Lazy: root + first level
            loadedNodes.clear();
            const rootChildren = parentChildMap.get('0') || [];
            const nodes = [], edges = [];

            if (nodeInfoMap.has('0')) {
                loadedNodes.add('0');
                const root = createNodeModel('0');
                if (root) nodes.push(root);
            }

            rootChildren.forEach(child => {
                loadedNodes.add(child.id);
                const m = createNodeModel(child.id);
                if (m) {
                    nodes.push(m);
                    edges.push({
                        source: '0',
                        target: child.id,
                        type: 'quadratic',
                        style: { stroke: '#722ed1', lineWidth: 2, endArrow: { path: 'M 0,0 L 6,-3 L 6,3 Z', fill: '#722ed1' } }
                    });
                }
            });

            graph.read({ nodes, edges });
            graph.fitView(60);
            updateStats();
            setStatus(`动态模式: ${loadedNodes.size}/${nodeInfoMap.size} 节点，点击展开`, 'success');
        } else {
            // Full load
            const nodes = [], edges = [];
            nodeInfoMap.forEach((_, id) => {
                const m = createNodeModel(id);
                if (m) nodes.push(m);
            });
            
            flatData.forEach(item => {
                if (item.parentId && item.parentId !== '0') {
                    edges.push({
                        source: item.parentId,
                        target: item.id,
                        type: 'quadratic',
                        style: { stroke: '#595959', lineWidth: 1.5, endArrow: { path: 'M 0,0 L 6,-3 L 6,3 Z', fill: '#595959' } }
                    });
                }
            });
            
            loadedNodes.clear();
            nodeInfoMap.forEach((_, id) => loadedNodes.add(id));
            
            graph.read({ nodes, edges });
            graph.fitView(60);
            updateStats();
            setStatus(`已加载全部 ${nodes.length} 节点`, 'success');
        }
    }

    function processFile(content) {
        if (!content) return;
        const mapping = {
            currentId: dom.configs.currentId.value.trim(),
            parentId: dom.configs.parentId.value.trim(),
            aInfo: dom.configs.aInfo.value.trim(),
            zInfo: dom.configs.zInfo.value.trim()
        };
        setLoading(true, '解析CSV...');
        setTimeout(() => {
            try {
                flatData = parseCSV(content, mapping);
                renderData(flatData);
                setLoading(false);
                lastContent = content;
            } catch (e) {
                setStatus('错误: ' + e.message, 'error');
                setLoading(false);
            }
        }, 50);
    }

    // Start
    initDOM();
    setupEventListeners();
    initGraph();
    setStatus('就绪 - 请选择 CSV 文件', 'normal');
</script>
